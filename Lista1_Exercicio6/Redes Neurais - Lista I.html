actAcc = 0
eixoX = 0
eixoY = 1
numeroNeuronios = 100
numeroCamadas = 10
melhorConfiguracao = {}
dados = []
for i in range(1, numeroNeuronios+1):
    for j in range(1, numeroCamadas+1):
        clf = MLPClassifier(hidden_layer_sizes=(i,)*j, max_iter=500, alpha=0.0001,
                             solver='sgd', verbose=0,  random_state=21,tol=0.01, activation='tanh')
        clf.fit(x_train, y_train)
        y_pred = clf.predict(x_test)
        tn, fp, fn, tp = confusion_matrix(y_test, y_pred).ravel()
        if fn > 0 and tp >0 and fp > 0 and tn > 0:
            #print(tn, fp, fn, tp)
            eixoYa , eixoXa = float(tp)/(float(fn)+float(tp)), float(tn)/(float(fp)+float(tn))
            #print(eixoXa, eixoYa)
            area1 = (eixoXa*eixoYa)/2.0
            area2 = (1-eixoXa)*(1-eixoYa)/2.0
            area3 = (1-eixoXa)*eixoYa
            areaTotal = area1 + area2 + area3
            curva = {
                "areaCurvaROC": areaTotal,
                "sensibilidade": eixoYa,
                "especificidade": eixoXa,
                "camadas": j,
                "neuronios": i
            }
            dados.append(curva)
            if areaTotal > actAcc:
                actAcc = areaTotal
                eixoX = eixoXa
                eixoY = eixoYa
                melhorConfiguracao = curva
                
print(melhorConfiguracao)
